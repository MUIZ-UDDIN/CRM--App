# Run these commands on your VPS to find remaining large files

# Find what's using space in /var/www
du -sh /var/www/* 2>/dev/null | sort -hr

# Check for large files in /var/www
find /var/www -type f -size +50M 2>/dev/null -exec ls -lh {} \; | awk '{print $5, $9}'

# Check /root again for hidden files
du -sh /root/.* 2>/dev/null | sort -hr | head -10

# Check system directories
du -sh /usr /opt /var 2>/dev/null
